  previous: T | null;
  isDirty: boolean;
}

class StateManager<T> {
  private state: ComponentState<T>;
  private listeners: ((state: T) => void)[] = [];

  constructor(initialState: T) {
    this.state = {
      current: initialState,
      previous: null,
      isDirty: false,
    };
  }

  public update(newState: Partial<T>): void {
    this.state.previous = { ...this.state.current };
    this.state.current = { ...this.state.current, ...newState };
    this.state.isDirty = true;
    this.notifyListeners();
  }

  public subscribe(listener: (state: T) => void): () => void {
    this.listeners.push(listener);
    return () => {
      this.listeners = this.listeners.filter(l => l !== listener);
    };
  }

  private notifyListeners(): void {
    this.listeners.forEach(listener => listener(this.state.current));

}
